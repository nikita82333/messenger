222 int getch_echo(bool echo=true){
223 struct termios oldt, newt;
224 int ch;
225 tcgetattr( STDIN_FILENO, &oldt );
226 newt = oldt;
227 newt.c_lflag &= ~ICANON;
228 if(echo)
229 newt.c_lflag &= ECHO;
230 else
231 newt.c_lflag &= ~ECHO;
232 tcsetattr( STDIN_FILENO, TCSANOW, &newt );
233 ch = getchar();
234 tcsetattr( STDIN_FILENO, TCSANOW, &oldt );
235 return ch;
236 }

246 int wherexy(int&x,int&y){
247 printf("\033[6n");
248 if(getch() != '\x1B') return 0;
249 if(getch() != '\x5B') return 0;
250 int in;
251 int ly = 0;
252 while((in = getch()) != ';')
253 ly = ly * 10 + in - '0';
254 int lx = 0;
255 while((in = getch()) != 'R')
256 lx = lx * 10 + in - '0';
257 x = lx;
258 y = ly;
259 }

273 int kbhit(){
274 struct termios oldt, newt;
275 int ch;
276 int oldf;
277 tcgetattr(STDIN_FILENO, &oldt);
278 newt = oldt;
279 newt.c_lflag &= ~(ICANON | ECHO);
280 tcsetattr(STDIN_FILENO, TCSANOW, &newt);
281 oldf = fcntl(STDIN_FILENO, F_GETFL, 0);
282 fcntl(STDIN_FILENO, F_SETFL, oldf | O_NONBLOCK);
283 ch = getchar();
284 tcsetattr(STDIN_FILENO, TCSANOW, &oldt);
285 fcntl(STDIN_FILENO, F_SETFL, oldf);
286 if(ch != EOF){
287 ungetc(ch, stdin);
288 return 1;
289 }
290 return 0;
291 }
